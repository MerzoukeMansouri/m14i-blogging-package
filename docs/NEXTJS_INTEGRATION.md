## Prerequisites

- Next.js 13+ (App Router) or Next.js 12+ (Pages Router)
- React 18+
- Tailwind CSS 3+
- Node.js 16+

## Installation

### 1. Install the Package

```bash
npm install m14i-blogging
# or
pnpm add m14i-blogging
# or
yarn add m14i-blogging
```

Modern package managers will automatically prompt you to install peer dependencies if needed.

If your package manager doesn't auto-install them:

```bash
npm install @hello-pangea/dnd react-markdown remark-gfm lucide-react
```

### 2. Install Tailwind CSS (if not already installed)

```bash
npm install -D tailwindcss postcss autoprefixer
npx tailwindcss init -p
```

### 3. Install shadcn/ui Components (only if using BlogBuilder editor)

```bash
npx shadcn@latest init
npx shadcn@latest add label input textarea select button card
```

**Note:** If you're only using the preview/display components, you can skip shadcn/ui.

## Setup

### App Router Setup

#### 1. Configure Tailwind CSS

Update `tailwind.config.ts` or `tailwind.config.js`:

```ts
import type { Config } from 'tailwindcss'

const config: Config = {
  content: [
    './pages/**/*.{js,ts,jsx,tsx,mdx}',
    './components/**/*.{js,ts,jsx,tsx,mdx}',
    './app/**/*.{js,ts,jsx,tsx,mdx}',
    // Add this line to scan the m14i-blogging package
    './node_modules/m14i-blogging/dist/**/*.{js,mjs,cjs}',
  ],
  theme: {
    extend: {},
  },
  plugins: [],
}
export default config
```

#### 2. Import Styles in Root Layout

**app/layout.tsx:**

```tsx
import type { Metadata } from 'next'
import { Inter } from 'next/font/google'
import 'm14i-blogging/styles'
import './globals.css'

const inter = Inter({ subsets: ['latin'] })

export const metadata: Metadata = {
  title: 'My Blog',
  description: 'Built with m14i-blogging',
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en">
      <body className={inter.className}>{children}</body>
    </html>
  )
}
```

#### 3. Create a Blog Editor Page (Admin)

**app/admin/blog/new/page.tsx:**

```tsx
'use client'

import { useState } from 'react'
import { BlogBuilder } from 'm14i-blogging'
import type { LayoutSection } from 'm14i-blogging'

export default function NewBlogPost() {
  const [sections, setSections] = useState<LayoutSection[]>([])
  const [title, setTitle] = useState('')

  const handleSave = async () => {
    // Save to your backend/database
    const post = {
      title,
      sections,
      slug: title.toLowerCase().replace(/\s+/g, '-'),
    }

    const response = await fetch('/api/blog', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(post),
    })

    if (response.ok) {
      alert('Blog post saved!')
    }
  }

  return (
    <div className="container mx-auto p-8">
      <input
        type="text"
        value={title}
        onChange={(e) => setTitle(e.target.value)}
        placeholder="Blog post title..."
        className="w-full mb-4 p-2 border rounded"
      />

      <BlogBuilder
        sections={sections}
        onChange={setSections}
        config={{
          callbacks: {
            onSave: handleSave,
          },
        }}
      />
    </div>
  )
}
```

#### 4. Create a Blog Display Page

**app/blog/[slug]/page.tsx:**

```tsx
import { BlogPreview, BlogSEO } from 'm14i-blogging'
import type { BlogPost, SEOConfig } from 'm14i-blogging'
import { generateBlogMetadata } from 'm14i-blogging'
import type { Metadata } from 'next'

// SEO Configuration
const seoConfig: SEOConfig = {
  siteUrl: 'https://yourdomain.com',
  siteName: 'Your Blog',
  defaultAuthor: {
    name: 'Your Name',
    url: 'https://yourdomain.com/about',
  },
  twitterSite: '@yourblog',
  enableStructuredData: true,
  autoGenerateDefaults: true,
}

// Fetch blog post (replace with your data fetching logic)
async function getBlogPost(slug: string): Promise<BlogPost> {
  // This is an example - replace with your actual data fetching
  const response = await fetch(`https://yourapi.com/blog/${slug}`)
  return response.json()
}

// Generate metadata for SEO
export async function generateMetadata({
  params
}: {
  params: { slug: string }
}): Promise<Metadata> {
  const post = await getBlogPost(params.slug)
  return generateBlogMetadata(post, seoConfig)
}

export default async function BlogPostPage({
  params
}: {
  params: { slug: string }
}) {
  const post = await getBlogPost(params.slug)

  return (
    <>
      {/* SEO Component with JSON-LD */}
      <BlogSEO post={post} config={seoConfig} />

      {/* Blog Content */}
      <article className="container mx-auto px-4 py-8">
        <BlogPreview
          title={post.title}
          sections={post.sections}
          showReadingTime={true}
          date={post.publishedDate}
        />
      </article>
    </>
  )
}
```

#### 3. Create a Blog Display Page

**pages/blog/[slug].tsx:**

```tsx
import { GetStaticProps, GetStaticPaths } from 'next'
import Head from 'next/head'
import { BlogPreview, BlogHead } from 'm14i-blogging'
import type { BlogPost, SEOConfig } from 'm14i-blogging'

const seoConfig: SEOConfig = {
  siteUrl: 'https://yourdomain.com',
  siteName: 'Your Blog',
  defaultAuthor: {
    name: 'Your Name',
    url: 'https://yourdomain.com/about',
  },
  twitterSite: '@yourblog',
  enableStructuredData: true,
  autoGenerateDefaults: true,
}

interface BlogPostPageProps {
  post: BlogPost
}

export default function BlogPostPage({ post }: BlogPostPageProps) {
  return (
    <>
      <Head>
        <BlogHead post={post} config={seoConfig} />
      </Head>

      <article className="container mx-auto px-4 py-8">
        <BlogPreview
          title={post.title}
          sections={post.sections}
          showReadingTime={true}
          date={post.publishedDate}
        />
      </article>
    </>
  )
}

export const getStaticPaths: GetStaticPaths = async () => {
  // Fetch all blog post slugs
  const response = await fetch('https://yourapi.com/blog/slugs')
  const slugs = await response.json()

  return {
    paths: slugs.map((slug: string) => ({ params: { slug } })),
    fallback: 'blocking',
  }
}

export const getStaticProps: GetStaticProps = async ({ params }) => {
  const response = await fetch(`https://yourapi.com/blog/${params?.slug}`)
  const post = await response.json()

  return {
    props: { post },
    revalidate: 60, // Revalidate every 60 seconds
  }
}
```

## Basic Usage

### Using Next.js Image Component

The package supports custom Image components for optimized images:

```tsx
import Image from 'next/image'
import { BlogPreview } from 'm14i-blogging'

export default function BlogPost({ post }) {
  return (
    <BlogPreview
      title={post.title}
      sections={post.sections}
      ImageComponent={Image}
    />
  )
}
```

**Method 1: CSS Variables (globals.css):**

```css

  --blog-font-family: 'Inter', sans-serif;
  --blog-radius-lg: 0.75rem;
}
```

```tsx

  classNames={{

  }}
/>
```

### Database Integration Example (Prisma)

**schema.prisma:**

```prisma
model BlogPost {
  id            String   @id @default(cuid())
  title         String
  slug          String   @unique
  excerpt       String?
  featuredImage String?
  sections      Json     // Store LayoutSection[] as JSON
  author        String
  tags          String[]
  category      String?
  publishedDate DateTime @default(now())
  updatedDate   DateTime @updatedAt
  published     Boolean  @default(false)
}
```

**lib/blog.ts:**

```tsx
import { PrismaClient } from '@prisma/client'
import type { BlogPost as BlogPostType } from 'm14i-blogging'

const prisma = new PrismaClient()

export async function createBlogPost(data: BlogPostType) {
  return await prisma.blogPost.create({
    data: {
      title: data.title,
      slug: data.slug,
      excerpt: data.excerpt,
      featuredImage: data.featuredImage,
      sections: data.sections,
      author: data.author?.name || 'Anonymous',
      tags: data.tags || [],
      category: data.category,
      publishedDate: data.publishedDate ? new Date(data.publishedDate) : new Date(),
    },
  })
}

export async function getBlogPost(slug: string) {
  return await prisma.blogPost.findUnique({
    where: { slug },
  })
}
```

## Troubleshooting

### CSS Not Loading

**Problem:** Styles are not applied to components.

**Solution:**
1. Make sure you've imported the styles: `import 'm14i-blogging/styles'`
2. Verify Tailwind config includes the package in `content` array
3. Clear Next.js cache: `rm -rf .next && npm run dev`

### TypeScript Errors

**Problem:** TypeScript can't find types.

**Solution:**
1. Ensure `moduleResolution` is set to `"bundler"` or `"node16"` in `tsconfig.json`
2. Try: `npm install --save-dev @types/react @types/react-dom`

### Drag and Drop Not Working

**Problem:** BlogBuilder drag and drop doesn't work.

**Solution:**
1. Ensure the component is client-side: add `'use client'` at the top of your file
2. Verify `@hello-pangea/dnd` is installed: `npm install @hello-pangea/dnd`

### shadcn/ui Components Not Found

**Problem:** "Module not found" errors for shadcn components.

**Solution:**
1. Install required components: `npx shadcn@latest add label input textarea select button card`
2. Make sure shadcn is properly initialized: `npx shadcn@latest init`

### Build Errors

**Problem:** Build fails with module errors.

**Solution:**
1. Clear cache and rebuild: `rm -rf .next node_modules && npm install && npm run build`
2. Check that all peer dependencies are installed
3. Ensure you're using Node.js 16+

### Images Not Optimized

**Problem:** Images are not using Next.js optimization.

**Solution:**
Pass the Next.js Image component:

```tsx
import Image from 'next/image'
import { BlogPreview } from 'm14i-blogging'

<BlogPreview ImageComponent={Image} {...props} />
```
